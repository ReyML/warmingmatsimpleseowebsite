# Add these rules to your Nginx server block (usually in /etc/nginx/sites-available/your-site)

# 1. Redirect index.html to root /
location = /index.html {
    return 301 $scheme://$host/;
}

# 3. Canonicalize directory URLs with a trailing slash.
# Place this inside `location /` before `try_files`. It only redirects an existing
# directory requested without its trailing slash; files and the site root are untouched.
if (-d $request_filename) {
    rewrite ^(.+[^/])$ https://www.warmingmat.shop$1/ permanent;
}
